@startuml

entity User {
    -id : Long
    -name : String
    -email : String
    -password : String
}

entity Client {
    -id : Long
    -name : String
}

entity Chantier {
    -id : Long
    -name : String
}


entity Intervention {
    -id : Long
}

entity ScheduledTask {
    -id : Integer
    -description : String
}

entity Week {
    -id : Integer
    -isHoliday : boolean
    -week_number : Integer
    -year : Integer
}

entity Typology {
    -id : Long
    -name : String
}

entity TypologyPool {
    -id : Long
}

User "1" -- "1" TypologyPool

TypologyPool "1" -- "0..*" Typology


/'
many to one
'/
Typology "1..*" -- "1..*" ScheduledTask

/'
A User can exist without any client and have many client
A client cannot exist without one and only one User
'/
User "1" -- "0..*" Client

/'
A client can exist without any chantier and have many chantiers
A chantier cannot exist without a client but can only have one max
'/
Client "1" -- "0..*" Chantier

/'
A chantier can exist before any intervention and have many interventions
An intervention is always tied to one chantier
'/
Chantier "1" -- "0..*" Intervention


/'
An intervention cannot exist without at least one InterventionWeek
An InterventionWeek cannot exist without exactly one intervention
'/
Intervention "1" -- "1..*" ScheduledTask

/'
A ScheduledTask cannot exist without at least one Week
a Week can exist without any ScheduleTask and have many
'/
ScheduledTask "0..*"  -- "1" Week

@enduml
